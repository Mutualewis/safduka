DROP EVENT purchases_summary_update;
delimiter |
CREATE EVENT purchases_summary_update
    ON SCHEDULE
	EVERY 30 MINUTE
        DO
BEGIN
DELETE FROM `ibero_db`.`purchases_summary_prsm`
WHERE date(created_at) = curdate();
INSERT INTO `ibero_db`.`purchases_summary_prsm`
(
`prsm_csn_season`,
`prsm_seller`,
`prsm_buyer`,
`prsm_wr_name`,
`prsm_weight`,
`prsm_month`,
`created_at`)
    select 
        
        `csn_season` ,
		`slr_name` ,
		`buyer` ,
		`wr_name` ,
		`weight` ,
		`month` ,
		CURRENT_TIMESTAMP
    from
        `purchases_summary` ;
      END |

delimiter ;
