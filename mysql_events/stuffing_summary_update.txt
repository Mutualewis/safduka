DROP EVENT stuffing_summary_update;
delimiter |
CREATE EVENT stuffing_summary_update
    ON SCHEDULE
	EVERY 30 MINUTE
        DO
BEGIN
DELETE FROM `ibero_db`.`stuffing_summary_stsm`
WHERE date(created_at) = curdate();
INSERT INTO `ibero_db`.`stuffing_summary_stsm`
(
stsm_buyer,
`stsm_year`,
`stsm_month`,
`stsm_weight`,
`created_at`)
    select 
        cb_name,
        `year` ,
		`month` ,
		`weight` ,
		CURRENT_TIMESTAMP
    from
        `stuffing_summary` `grs`;
      END |

delimiter ;
